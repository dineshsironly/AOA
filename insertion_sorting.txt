#include<stdio.h>

int *insertion(int arr[], int n)
{
    int temp;
    for (int i  = 1; i < n; i++)
    {
        for (int j = i-1; j>=0; j--)
        {
            if (arr[j] > arr[j+1])
            {
                temp = arr[j+1];
                arr[j+1] = arr[j];
                arr[j] = temp;
            }
            else{
                break;
            }
        }
    }
    return arr;
}

int main()
{
    int arr[] = {20, 30, 50, 90, 70, 10, 60, 40, 80};
    
    int n = sizeof(arr)/sizeof(arr[0]);

    insertion(arr, n);

    printf("Insertion Sorting\n");
    printf("The sorted array is :");
    for (int i = 0; i<n; i++)
    {
        printf("%d ", arr[i]);
    }
    return 0;
}
